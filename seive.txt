vector<ll> primes;

bool dataset[1000000]={false};
void seive()
{
    //primes.reserve(1000000);
    dataset[0]=true;
    primes.push_back(2);
    for(int i=3;i*i<1000000;i+=2)
        if(!dataset[i])
            for(int j=i*i;j<1000000;j+=i*2)
                dataset[j]=true;
    for(int i=3;i<1000000;i+=2)
    {
        if(!dataset[i])
        primes.push_back(i);
    }
    //cout<<primes.size()<<endl;
}
